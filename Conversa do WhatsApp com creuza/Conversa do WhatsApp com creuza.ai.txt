from twilio.rest import Client
from twilio.twiml.messaging_response import MessagingResponse

# Tokens da Twilio
ACCOUNT_SID = 'seu_account_sid'
AUTH_TOKEN = 'seu_auth_token'
CLIENT = Client(ACCOUNT_SID, AUTH_TOKEN)

# Número do WhatsApp
FROM_WHATSAPP_NUMBER = 'seu_numero_whatsapp'

def resposta_mensagem(mensagem):
    # Lógica para responder
    resposta = 'Olá! Como posso ajudar?'
    return resposta

def enviar_resposta(mensagem, resposta):
    message = CLIENT.messages.create(
        from_=FROM_WHATSAPP_NUMBER,
        body=resposta,
        to=mensagem.from_
    )
    return message.sid

def receber_mensagem(request):
    mensagem = request.values.get('Body', None)
    resposta = resposta_mensagem(mensagem)
    enviar_resposta(mensagem, resposta)
    return 'Mensagem enviada!'

# Flask
from flask import Flask, request
app = Flask(__name__)

@app.route('/whatsapp', methods=['POST'])
def whatsapp():
    return receber_mensagem(request)

if __name__ == '__main__':
    app.run()
